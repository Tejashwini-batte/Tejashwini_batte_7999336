import java.net.*;
class Server{
    public static void main(String[] args){
        try{
            ServerSocket ss=new ServerSocket(5000);
            System.out.println("Waiting");
            Socket s = ss.accept();
            System.out.println("Connected");
            ss.close();
        }
        catch(Exception e){
            System.out.println(e);
}
}
}
