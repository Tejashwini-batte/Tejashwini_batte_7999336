class Car{
String make;
String model;
int year;
void displayDetails()
{
System.out.println(make);
System.out.println(model);
System.out.println(year);
}
public static void main(String[] args)
Car c1 = new Car();
c1.make = "Hyundai";
c1.model = "i20";
c1.year = 2022;
c1.displayDetails();
}
}
